var content = document.getElementById("model");
var button = document.getElementById("show-more")

button.onclick = function(){

	if(content.className == "open"){
		content.className = "";
		button.innerHTML = "More";

	} else {
		content.className = "open";
		button.innerHTML = "Less";
	}

};
